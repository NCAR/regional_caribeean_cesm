Define the resolution dz(z)#

dz = np.zeros(ds.depth.shape)

dz[0] = 2.0 * ds.depth[0]
print('depth = {}; dz = {}'.format(ds.depth[0].values, dz[0]))
for k in range(1,len(dz)):
    dz[k] = 2.0 * (ds.depth[k] - dz[0:k].sum())
    print('depth = {}; dz = {}'.format(ds.depth[k].values, dz[k]))
